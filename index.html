Git

My finetuning for Git

Meld
Tool I use to solve Merge conflict with Git, my config 
        [diff]
            tool = meld
        [difftool]
            prompt = false
        [difftool "meld"]
            cmd = meld "$LOCAL" "$REMOTE"
        [merge]
            tool = meld
        [mergetool]
            # don't ask if we want to skip merge
            prompt = false
            # don't create backup *.orig files
            keepBackup = false
